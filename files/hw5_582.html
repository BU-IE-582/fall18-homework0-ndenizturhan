Musk Data The task is transforming the data from instance-level to bag-level based on Multiple Instance Learning (MIL). To this aim, the data should be clustered firstly and then transforming to bag-level representation. Two distance measures are used namely Euclidean and Manhattan. Also, two different clustering algorithm are used for each distance measure: Pam (modified k-means) and Hierarchical. As the next step, lasso logistic regression model is trained for classifying the bags. For each bags created by clustering, a model is trained and the one with the best parameters is chosen. Lastly, 10-fold cross validation is done on the chosen model.

firstIter=TRUE
for(k in 1:nofreplication){
  for(j in 1:noffold) {
    testindices=cvindices[[k]][[j]]
    lambda1 <- optL1(bags$V1[-testindices], penalized = ~., data = bags[-testindices,-1], 
                     fusedl = FALSE, minlambda1 = 0, maxlambda1 = 2, 
                     fold = 10)
    
    pen <- penalized(bags$V1[-testindices], penalized = ~., data = bags[-testindices,-1], 
                     lambda1 = lambda1$lambda, 
                     fusedl = FALSE)
    lassopredict<-predict(pen,penalized=bags[testindices,-1])
    
    
    lassopredict[lassopredict>0.5]=1
    lassopredict[lassopredict<=0.5]=0
    
    confusion.matrix <- table(bags$V1[testindices], lassopredict)
    print(confusion.matrix)
    accuracy.percent <- 100*sum(diag(confusion.matrix))/sum(confusion.matrix)
    print(paste("accuracy1:",accuracy.percent,"%"))
    a<-roc(bags$V1[testindices],lassopredict)
    roctut<-a$auc
    plot(a)
    #aggregation of results, if it is first iteration, initialize the data table
    if(firstIter){
      overallresults=data.table(Rep=k,Fold=j,Method="Lasso",Accuracy=accuracy.percent)
      overallroc=data.table(Rep=k,Fold=j,Method="Lasso",ROC=roctut)
      firstIter=FALSE
    } else {
      temp=data.table(Rep=k,Fold=j,Method="Lasso",Accuracy=accuracy.percent)
      overallresults=rbind(overallresults,temp)
      temp2=data.table(Rep=k,Fold=j,Method="Lasso",ROC=roctut)
      overallroc=rbind(overallroc,temp2)
    }
  }}
